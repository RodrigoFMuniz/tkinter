# Tkinter

## Importando o Tkinter

    import tkinter

## Instanciando a primeira janela

    window = tk.Tk()

## Executando a primeira janela

    window.mainloop()

## Propriedades da janela -> Title

    window.title('Notepad')

## Propriedades da janela -> minsize

    window.minsize(width=780, height=520)
